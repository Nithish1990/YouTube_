



interface ChannelControllerFactory {
    public ChannelController createController(Channel model, ChannelView view);
}

class ContentCreatorControllerFactory implements ChannelControllerFactory {
    @Override
    public ChannelController createController(Channel model, ChannelView view) {
        return new ChannelController(model, view) {
            @Override
            public void updateView() {
                super.updateView();
                view.displayUploadButton();
            }
        };
    }
}

class AdminControllerFactory implements ChannelControllerFactory {
    @Override
    public ChannelController createController(Channel model, ChannelView view) {
        return new ChannelController(model, view) {
            @Override
            public void updateView() {
                super.updateView();
                view.displayMonetizationOption();
                view.displayDeleteOption();
            }
        };
    }
}

class Main {
    public static void main(String[] args) {
        Channel channel = new Channel("My Channel", "My channel description");
        channel.addVideo("Video 1");
        channel.addVideo("Video 2");
        ChannelView view = new ChannelPageView();
        String userType = "admin";
        ChannelControllerFactory factory;
        if(userType.equals("contentCreator")){
            factory = new ContentCreatorControllerFactory();
        }else{
            factory = new AdminControllerFactory();
        }
        ChannelController controller = factory.createController(channel, view);
        controller.updateView();
    }
}
